\newenvironment{tzfigure}[1]
{
    \begin{figure}[ht]
        \centering
        #1
        \begin{tikzpicture}
}
{

}

{
    \begin{figure}[ht]
        \centering

}

    \end{tikzpicture}   
\end{figure}
}

{
    \begin{subfigure}[ht]{#2}

}
{
    \end{tikzpicture}   
\end{subfigure}
